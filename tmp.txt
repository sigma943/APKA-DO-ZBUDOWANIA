import { NextResponse } from 'next/server';
import WebSocket from 'ws';
import path from 'path';
import fs from 'fs';

export const dynamic = 'force-dynamic';

export interface BusLocation {
  id: string; // Internal vehicle ID
  routeId: string; // e.g., "108", "219" (Line number)
  name: string; // Used by frontend, e.g. "MKS 108"
  routeShortName: string; // Used by frontend, e.g. "108"
  lat: number;
  lon: number;
  speed?: number; 
  direction?: string; // Destination or route name
  delay?: number; // Delay in seconds
  dataAgeSec?: number;
  schedule?: { id: number; name: string; planned: string; real: string; }[];
  routePath?: number[];
  model?: string; // E.g. "Autosan", "Iveco"
}

// Global caches and singletons
const globalVehicles: Record<string, any> = (globalThis as any)._mksVehicles || {};
(globalThis as any)._mksVehicles = globalVehicles;

let stopsDict: Record<string, string> = {};
try {
  const p = path.join(process.cwd(), 'app', 'api', 'vehicles', 'stops-dictionary.json');
  if (fs.existsSync(p)) {
     stopsDict = JSON.parse(fs.readFileSync(p, 'utf8'));
  }
} catch(e) {
  console.log('Could not load stops dictionary');
}

// Background Worker State
let lastSnapshotFetch = (globalThis as any)._mksLastFetch || 0;

async function fetchSnapshot(force = false) {
  const now = Date.now();
  if (!force && now - lastSnapshotFetch < 10000) return; // 10 seconds throttle in background
  lastSnapshotFetch = now;
  (globalThis as any)._mksLastFetch = now;

  try {
    const res = await fetch('http://185.214.67.112/api/its/vehicles', { 
      cache: 'no-store',
      headers: { 'Accept': 'application/json', 'Host': 'einfo.zgpks.rzeszow.pl' },
      signal: AbortSignal.timeout(5000)
    });
    
    if (!res.ok) return;

    const data = await res.json();
    const vehicles = Array.isArray(data) ? data : (data.vehicles || data.items || data.data || []);
    if (vehicles && Array.isArray(vehicles)) {
      for (const item of vehicles) {
        const id = String(item.vehicle_id || item.id);
        if (!id) continue;
        
        if (!globalVehicles[id]) {
          globalVehicles[id] = { id, lastSeen: 0 };
        }
        
        const lastWsUpdate = globalVehicles[id].lastSeenWs || 0;
        const isFromWsRecently = (now - lastWsUpdate < 15000); 

        const lat = item.lat || item.latitude || item.y;
        const lon = item.lng || item.long || item.longitude || item.x;
        
        if (lat && lon && (!isFromWsRecently || !globalVehicles[id].position)) {
          globalVehicles[id].position = {
            lat: parseFloat(String(lat)),
            long: parseFloat(String(lon)),
            speed: parseFloat(String(item.speed || 0)),
            course: parseFloat(String(item.course || 0))
          };
          globalVehicles[id].lastSeen = now;
        }

        const line = String(item.line_name || item.route_short_name || item.routeId || '').trim();
        if (line && line !== '---' && line !== 'null' && line !== 'undefined') {
          globalVehicles[id].lastKnownLine = line;
          globalVehicles[id].lastJourneyUpdate = now;
          if (!globalVehicles[id].journey) {
            globalVehicles[id].journey = { 
              line: { line_name: line, name: line },
              route: { description: item.route_name || item.destination || '', name: item.route_name || '' }
            };
          }
        }
        
        if (item.model_name || item.model) {
          globalVehicles[id].apiModel = item.model_name || item.model;
        }
      }
    }
  } catch (e) {}
}

let ws: WebSocket | null = (globalThis as any)._mksWs || null;

let backgroundServiceStarted = false;

function initBackgroundService() {
  if (backgroundServiceStarted) return;
  backgroundServiceStarted = true;
  if ((globalThis as any)._mksWorkerRunning) return;
  (globalThis as any)._mksWorkerRunning = true;

  console.log('MKS Background Service Starting...');

  const wsLoop = () => {
    if (ws && (ws.readyState === WebSocket.OPEN || ws.readyState === WebSocket.CONNECTING)) {
      const lastMsg = (globalThis as any)._mksWsTime || 0;
      if (Date.now() - lastMsg > 30000) {
        try { ws.close(); } catch(e) {}
        ws = null;
      } else {
        return;
      }
    }

    try {
      ws = new WebSocket('ws://185.214.67.112:3000/rist');
      (globalThis as any)._mksWs = ws;
      (globalThis as any)._mksWsTime = Date.now();

      ws.on('message', (data) => {
        try {
          (globalThis as any)._mksWsTime = Date.now();
          const parsed = JSON.parse(data.toString());
          const id = parsed.vehicle_id;
          if (!id) return;
          if (!globalVehicles[id]) globalVehicles[id] = { id, lastSeen: 0 };
          
          globalVehicles[id].lastSeen = Date.now();
          globalVehicles[id].lastSeenWs = Date.now();
          
          if (parsed.journey !== undefined && parsed.journey !== null) {
              globalVehicles[id].journey = parsed.journey;
              globalVehicles[id].lastKnownLine = parsed.journey?.line?.line_name || parsed.journey?.line?.name;
              globalVehicles[id].lastJourneyUpdate = Date.now();
          }
          if (parsed.position) globalVehicles[id].position = parsed.position;
          if (parsed.delay !== undefined) globalVehicles[id].delay = parsed.delay;
          if (parsed.next_stop_points) globalVehicles[id].next_stop_points = parsed.next_stop_points;
        } catch (e) {}
      });
      ws.on('close', () => { ws = null; (globalThis as any)._mksWs = null; });
      ws.on('error', () => { ws?.close(); });
    } catch (e) {}
  };

  setInterval(wsLoop, 5000);
  setInterval(fetchSnapshot, 10000);
  fetchSnapshot();
}

// Lazy initialized in GET

function formatStopName(rawName: string): string {
  if (!rawName) return 'Przystanek nieznany';
  
  const isRzeszow = rawName.includes('Rzeszów');
  const isDA = rawName.toLowerCase().includes('d.a.') || rawName.toLowerCase().includes('dworzec');
  
  // Try to extract the stop code from the end of the name (e.g., " 01" or " st. 01")
  const words = rawName.trim().split(' ');
  let code = '';
  let baseName = rawName;

  if (words.length > 1) {
    const lastWord = words[words.length - 1];
    const secondLastWord = words[words.length - 2];

    if (/^\d+$/.test(lastWord)) {
      code = lastWord;
      if (secondLastWord === 'st.') {
        baseName = words.slice(0, -2).join(' ');
      } else {
        baseName = words.slice(0, -1).join(' ');
      }
    }
  }

  if (isRzeszow) {
    if (code) {
      if (isDA) {
        // User requested: No platform info (st.) for Rzeszów D.A. in the panel/list
        return baseName.replace(/\s+st\.$/, '').trim();
      } else {
        return `${baseName} ${code}`;
      }
    }
    return rawName;
  } else {
    // Non-Rzeszów: remove the code
    return baseName.trim();
  }
}

export async function GET(request: Request) {
  initBackgroundService();
  try {
    const { searchParams } = new URL(request.url);
    const inactiveParam = searchParams.get('inactive') === 'true';
    const forceParam = searchParams.get('force') === 'true';
    
    const now = Date.now();
    
    // Check if WebSocket needs to be revived synchronously
    if (forceParam || now - ((globalThis as any)._mksWsTime || 0) > 30000) {
       (globalThis as any)._mksWsTime = now; // Prevent multiple concurrent revivals
       if ((globalThis as any)._mksWs) {
         try { (globalThis as any)._mksWs.close(); } catch(e) {}
       }
       (globalThis as any)._mksWs = null;
       
       // Force WS connection immediately instead of waiting for setInterval
       try {
         const ws = new WebSocket('ws://185.214.67.112:3000/rist');
         (globalThis as any)._mksWs = ws;
         ws.on('message', (data) => {
           try {
             (globalThis as any)._mksWsTime = Date.now();
             const parsed = JSON.parse(data.toString());
             const id = parsed.vehicle_id;
             if (!id) return;
             if (!globalVehicles[id]) globalVehicles[id] = { id, lastSeen: 0 };
             
             globalVehicles[id].lastSeen = Date.now();
             globalVehicles[id].lastSeenWs = Date.now();
             
             if (parsed.journey !== undefined && parsed.journey !== null) {
                 globalVehicles[id].journey = parsed.journey;
                 globalVehicles[id].lastKnownLine = parsed.journey?.line?.line_name || parsed.journey?.line?.name;
                 globalVehicles[id].lastJourneyUpdate = Date.now();
             }
             if (parsed.position) globalVehicles[id].position = parsed.position;
             if (parsed.delay !== undefined) globalVehicles[id].delay = parsed.delay;
             if (parsed.next_stop_points) globalVehicles[id].next_stop_points = parsed.next_stop_points;
           } catch (e) {}
         });
         ws.on('close', () => { (globalThis as any)._mksWs = null; });
         ws.on('error', () => {});
       } catch (e) {}
    }

    if (forceParam || now - lastSnapshotFetch > 15000) {
       await fetchSnapshot(true);
    }
    
    const mapped: BusLocation[] = [];

    const vehicleArray = Object.values(globalVehicles);
    for (const v of vehicleArray) {
      if (!v.position || v.position.lat == null || v.position.long == null) continue;
      
      // Filter out stale vehicles (not seen in the last 15 minutes)
      if (now - v.lastSeen > 15 * 60 * 1000) {
        delete globalVehicles[v.id];
        continue;
      }
      
      const lineName = v.lastKnownLine || v.journey?.line?.line_name || v.journey?.line?.name || '---';
      
      const isDepotArea = (v.position.lat > 50.04 && v.position.lat < 50.05 && v.position.long > 21.99 && v.position.long < 22.01);
      
      if (!inactiveParam) {
        if (lineName === '---' || lineName === 'null') {
           const wasRecentlyActive = (now - (v.lastJourneyUpdate || 0) < 5 * 60000);
           if (!wasRecentlyActive) continue;
        }
        if (v.position.speed > 85 && !isDepotArea) continue;
      }
      
      const isMoving = v.position.speed > 5;
      const destination = v.journey?.route?.description || v.journey?.route?.name || (isMoving ? 'W trasie' : 'Postój');
      const ageSec = Math.floor((now - v.lastSeen) / 1000);

      const schedule = v.next_stop_points?.map((sp: any) => ({
        id: sp.stop_point_id,
        name: formatStopName(stopsDict[sp.stop_point_id]),
        planned: sp.planned_departure_time ? sp.planned_departure_time.replace(' ', 'T') : null,
        real: sp.real_departure_time ? sp.real_departure_time.replace(' ', 'T') : null
      })) || [];

      mapped.push({
        id: String(v.id),
        routeId: lineName,
        name: `MKS ${lineName !== '---' ? lineName : v.id}`,
        routeShortName: lineName !== '---' ? lineName : '?',
        lat: v.position.lat,
        lon: v.position.long,
        speed: v.position.speed,
        direction: destination,
        delay: v.delay,
        dataAgeSec: ageSec,
        schedule,
        routePath: v.journey?.route?.stop_points?.map((sp: any) => sp.stop_point_id) || []
      });
    }

    const resultJson = JSON.stringify(mapped);
    
    // Generate simple stable hash for ETag
    let hash = 0;
    for (let i = 0; i < resultJson.length; i++) {
       hash = Math.imul(31, hash) + resultJson.charCodeAt(i) | 0;
    }
    const etag = `W/"${Math.abs(hash).toString(16)}"`;
    
    if (request.headers.get('if-none-match') === etag) {
       return new NextResponse(null, { status: 304, headers: { 'ETag': etag, 'Cache-Control': 'no-cache, must-revalidate' } });
    }

    return new NextResponse(resultJson, {
      headers: {
        'Content-Type': 'application/json',
        'Cache-Control': 'no-cache, must-revalidate',
        'ETag': etag
      },
    });

  } catch (error) {
    console.error('Error in vehicle API:', error);
    return NextResponse.json({ error: 'Internal error' }, { status: 500 });
  }
}
